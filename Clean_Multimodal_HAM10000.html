import torch
from torch.utils.data import Dataset
from PIL import Image, UnidentifiedImageError
from torchvision import transforms
import PIL
import numpy as np

class MultimodalSkinLesionDataset(Dataset):
    def __init__(self, dataframe, img_dir, transform=None):
        self.data = dataframe.copy()
        self.img_dir = img_dir
        self.transform = transform

        # Categorical and numerical features
        self.categorical_features = ['sex', 'localization']
        self.numerical_features = ['age']

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        # Load image
        # we use .iloc for positional indexing because it is required by pandas series
        image_id = self.data.iloc[idx]['image_id']
        img_name = f"{image_id}.jpg"
        img_path = os.path.join(self.img_dir, img_name)

        try:
          image = Image.open(img_path).convert('RGB')
        except (UnidentifiedImageError, FileNotFoundError):  # Catch both errors
          print(f"Warning: Could not open image {img_path}. Returning black placeholder.")
          # Create a black placeholder image:
          image = Image.fromarray(np.zeros((224, 224, 3), dtype=np.uint8)) # Adjust size if needed

        if self.transform:
            image = self.transform(image)

        # Get categorical features
        categorical_series = self.data.iloc[idx][self.categorical_features]
        categorical_values = categorical_series.values
        categorical_values = categorical_values.astype(int)
        categorical = torch.tensor(
            categorical_values,
            dtype=torch.long
        )

        numerical_series = self.data.iloc[idx][self.numerical_features]
        numerical_values = numerical_series.values
        # print(f"numerical values type: {type(numerical_values)}")
        # print(f"numerical values values type: {numerical_values.dtype}")

        numerical_values = numerical_values.astype(float)
        # print(f"numerical values type: {type(numerical_values)}")
        # print(f"numerical values values type: {numerical_values.dtype}")

        # Get numerical features
        numerical = torch.tensor(
            numerical_values,
            dtype=torch.float
        )

        # Get label
        label = torch.tensor(
            self.data.iloc[idx]['dx'].astype(int),
            dtype=torch.long
        )

        return image, categorical, numerical, label

import torch.nn as nn
from torchvision import models

class ClinicalFeatureExtractor(nn.Module):
    def __init__(self, embedding_sizes, num_numerical):
        super().__init__()
        # Initialize embedding layers for each categorical feature
        self.embeddings = nn.ModuleList([
            nn.Embedding(num_categories, embed_dim)
            for num_categories, embed_dim in embedding_sizes
        ])

        # Fully connected layers
        total_embedding_dim = sum([embed_dim for _, embed_dim in embedding_sizes])
        self.fc = nn.Sequential(
            nn.Linear(total_embedding_dim + num_numerical, 512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Dropout(0.3)
        )

    def forward(self, categorical, numerical):
        # Embed each categorical feature
        embedded = [emb(categorical[:, i]) for i, emb in enumerate(self.embeddings)]
        embedded = torch.cat(embedded, dim=1)  # [batch_size, total_embedding_dim]

        # Concatenate with numerical features
        combined = torch.cat([embedded, numerical], dim=1)  # [batch_size, total_embedding_dim + num_numerical]

        # Pass through FC layers
        return self.fc(combined)  # [batch_size, 256]

class MultimodalSkinLesionClassifierV5(nn.Module):
    '''
    Location  where best model parameters are saved
    model_dir = '/content/drive/My Drive/Capstone/Models/Multimodal/V5'
    '''

    def __init__(self, embedding_sizes, num_numerical, num_classes=6):
        super(MultimodalSkinLesionClassifierV5, self).__init__()

        resnet50 = models.resnet50(weights='IMAGENET1K_V1')

        #first unfreeze all parameters
        for param in resnet50.parameters():
          param.requires_grad = True

        # Replace final Fully connected (FC) layer with nn.Identity() layer to extract the final highly dimensional feature vector
        resnet50.fc = nn.Identity()

        # Image Branch: ResNet18
        self.image_model = resnet50

        # Clinical Features Branch
        self.clinical_extractor = ClinicalFeatureExtractor(embedding_sizes, num_numerical)

        # Neural Network Based Fusion + Classification head (final FC)
        self.classifier = nn.Sequential(
            nn.Linear(2304, 512), # 2048 image + 256 clinical data
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(128, num_classes)
        )

    def forward(self, image, categorical, numerical):
        # Image features
        img_features = self.image_model(image)  # [batch_size, 2048]

        # Clinical features
        clinical_features = self.clinical_extractor(categorical, numerical)  # [batch_size, 256]

        # Combine features
        combined = torch.cat((img_features, clinical_features), dim=1)  # [batch_size, 2304]

        # Fusion + Classification
        logits = self.classifier(combined)  # [batch_size, num_classes]

        return logits

import time
from sklearn.metrics import f1_score, precision_score, recall_score
from torch.utils.tensorboard import SummaryWriter
def train_model(model, criterion, optimizer, scheduler, num_epochs=50, patience=10, delta=0.01):
    since = time.time()

    #location to save best model parameters
    model_dir = '/content/drive/My Drive/Capstone/Models/Multimodal/V8'
    os.makedirs(model_dir, exist_ok=True)
    best_model_params_path = os.path.join(model_dir, 'best_model_params.pt')
    print(f"Best model parameters will be saved to: {best_model_params_path}")

    torch.save(model.state_dict(), best_model_params_path)

    # Initialize TensorBoard SummaryWriter
    tensorboard_log_dir = '/content/drive/My Drive/Capstone/TensorBoardLogs/Multimodal/V8'
    os.makedirs(tensorboard_log_dir, exist_ok=True)
    writer = SummaryWriter(tensorboard_log_dir)

    # Log the model graph (experiment later)

    # Early Stopping vars
    best_acc = 0.0
    best_loss = float('inf')
    best_f1 = 0.0
    epochs_no_improve = 0
    early_stop = False

    for epoch in range(num_epochs):
        if early_stop:
          print('Early stopping triggered. Training stopped.')
          break #break the epoch loop

        print(f'Epoch {epoch}/{num_epochs - 1}')
        print('-' * 10)

        # Each epoch has a training and validation phase
        for phase in ['train', 'val']:
            if phase == 'train':
                model.train()  # Set model to training mode
            else:
                model.eval()   # Set model to evaluate

            # variables used to compute metrics in validation phase
            running_loss = 0.0
            running_corrects = 0
            all_labels = []
            all_preds = []

            # Iterate over data
            for images, categoricals, numericals, labels in dataloaders[phase]:
                images = images.to(device)
                categoricals = categoricals.to(device)
                numericals = numericals.to(device)
                labels = labels.to(device)

                # zero the parameter gradients
                optimizer.zero_grad()

                # forward
                with torch.set_grad_enabled(phase == 'train'):
                  outputs = model(images, categoricals, numericals)
                  _, preds = torch.max(outputs, 1)
                  loss = criterion(outputs, labels)

                  # backpropagation
                  if phase == 'train':
                    # compute gradients
                    loss.backward()
                    # optimize (step in the gradient direction that results in the lower loss)
                    optimizer.step()

                # statistics
                running_loss += loss.item() * images.size(0)
                running_corrects += torch.sum(preds == labels.data)

                # Collect all labels and predictions for metrics computation
                # be careful: i cant convert a pytorch tensor on the gpu to numpy array without moving it first to cpu
                all_labels.extend(labels.cpu().numpy())
                all_preds.extend(preds.cpu().numpy())
            if phase == 'train':
                scheduler.step()

            epoch_loss = running_loss / dataset_sizes[phase]
            epoch_acc = running_corrects.double() / dataset_sizes[phase]
            epoch_f1 = f1_score(all_labels, all_preds,average='macro', zero_division=0)

            print(f'{phase} Loss: {epoch_loss:.4f} Acc: {epoch_acc:.4f} F1: {epoch_f1:.4f}')

            # Log metrics to TensorBoard
            writer.add_scalar(f'{phase}/Loss', epoch_loss, epoch)
            writer.add_scalar(f'{phase}/Accuracy', epoch_acc, epoch)
            writer.add_scalar(f'{phase}/F1', epoch_f1, epoch)

            if phase == 'val':
              #compute metrics
              epoch_precision = precision_score(all_labels, all_preds, average='macro', zero_division=0)
              epoch_recall = recall_score(all_labels, all_preds, average='macro', zero_division=0)
              epoch_f1 = f1_score(all_labels, all_preds, average='macro', zero_division=0)

              # Log the metrics to TensorBoard
              writer.add_scalar(f'{phase}/Precision', epoch_precision, epoch)
              writer.add_scalar(f'{phase}/Recall', epoch_recall, epoch)
              writer.add_scalar(f'{phase}/F1_Score', epoch_f1, epoch)

              # Early stopping logic
              if epoch_loss < best_loss - delta:
                  best_loss = epoch_loss
                  best_acc = epoch_acc
                  best_f1 = epoch_f1
                  torch.save(model.state_dict(), best_model_params_path)
                  epochs_no_improve = 0
              else:
                  epochs_no_improve += 1
                  print(f'EarlyStopping counter: {epochs_no_improve} out of {patience}')

                  if epochs_no_improve >= patience:
                      print('Early stopping')
                      early_stop = True
                      break  # Break the validation loop

        print()

    time_elapsed = time.time() - since
    print(f'Training complete in {time_elapsed // 60:.0f}m {time_elapsed % 60:.0f}s')
    print(f'Best val Loss: {best_loss:.4f} Acc: {best_acc:4f} F1:{best_f1:4f}' )

    # load best model weights
    model.load_state_dict(torch.load(best_model_params_path, weights_only=True))

def test_model(model):
  # Set model to evaluation mode
  model.eval()

  # Initialize variables
  running_corrects = 0
  all_preds = []
  all_labels = []

  # Evaluation loop
  with torch.no_grad():
      for images, categoricals, numericals, labels in dataloaders['test']:
          images = images.to(device)
          categoricals = categoricals.to(device)
          numericals = numericals.to(device)
          labels = labels.to(device)
          #outputs is a tensor that contains the raw logits or scores for each class
          outputs = model(images, categoricals, numericals)
          # Get predictions by selecting the class with the highest score
          #we ignore max_value but retrieve the indice (preds) which correspond to the class label that the model predicted
          _, preds = torch.max(outputs, 1)

          running_corrects += torch.sum(preds == labels.data)
          all_preds.extend(preds.cpu().numpy())
          all_labels.extend(labels.cpu().numpy())

  # Calculate accuracy
  test_accuracy = running_corrects.double() / dataset_sizes['test']
  print(f'Test Accuracy: {test_accuracy:.4f}')

  # Generate classification report
  report = classification_report(all_labels, all_preds, target_names=class_names)
  print(report)

from sklearn.utils.class_weight import compute_class_weight
import numpy as np
import torch

# Define label mapping
label_mapping = {'bkl': 0, 'nv': 1, 'df': 2, 'mel': 3, 'vasc': 4, 'bcc': 5, 'akiec': 6}

# Define classes in the correct order
classes = np.array(list(label_mapping.values()))  # [0, 1, 2, 3, 4, 5, 6]
print("Classes:", classes)

# Compute class weights based on training data and computes class weights that are inversely proportional to class counts in training dataset
class_weights_auto = compute_class_weight(
    class_weight='balanced',
    classes=classes,
    y=train_labels
)

# Convert to NumPy array
class_weights_list = np.array(class_weights_auto)
print("Original Class Weights:", class_weights_list)

# Normalize weights to sum to the number of classes (optional but recommended)
total_classes = len(classes)
sum_weights = class_weights_list.sum()
class_weights_normalized = class_weights_list * (total_classes / sum_weights)
print("Normalized Class Weights:", class_weights_normalized)

# Convert to tensor and move to device
class_weights_tensor = torch.tensor(class_weights_normalized, dtype=torch.float).to(device)
